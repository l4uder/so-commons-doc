46 t_log* log_create(char* file, char *program_name, bool is_active_console, t_log_level detail) {
47  t_log* logger = malloc(sizeof(t_log));
48 
49  if (logger == NULL) {
50  error_show("Cannot create logger");
51  return NULL;
52  }
53 
54  FILE *file_opened = NULL;
55 
56  if (file != NULL) {
57  if (!string_ends_with(file, ".log")) {
58  error_show("Log file must have .log extension");
59  free(logger);
60  return NULL;
61  }
62  file_opened = txt_open_for_append(file);
63 
64  if (file_opened == NULL) {
65  error_show("Cannot create/open log file");
66  free(logger);
67  return NULL;
68  }
69  }
70 
71  logger->file = file_opened;
72  logger->is_active_console = is_active_console;
73  logger->detail = detail;
74  logger->pid = process_getpid();
75  logger->program_name = string_duplicate(program_name);
76  return logger;
77 }

140 static void _log_write_in_level(t_log* logger, t_log_level level, const char* message_template, va_list list_arguments) {
141  if (!_is_level_enabled(logger, level)) {
142  return;
143  }
144 
145  char *message = string_from_vformat(message_template, list_arguments);
146  char *time = temporal_get_string_time("%H:%M:%S:%MS");
147  unsigned int thread_id = process_get_thread_id();
148 
149  char *buffer = string_from_format("[%s] %s %s/(%d:%d): %s\n",
150  log_level_as_string(level),
151  time,
152  logger->program_name,
153  logger->pid,
154  thread_id,
155  message);
156 
157  if (logger->file != NULL) {
158  txt_write_in_file(logger->file, buffer);
159  }
160 
161  if (logger->is_active_console) {
162  char *console_buffer = string_from_format("%s%s%s",
163  log_level_color(level),
164  buffer,
165  reset_color);
166  txt_write_in_stdout(console_buffer);
167  free(console_buffer);
168  }
169 
170  free(time);
171  free(message);
172  free(buffer);
173 }

175 static bool _is_level_enabled(t_log* logger, t_log_level level) {
