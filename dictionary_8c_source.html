46 static unsigned int dictionary_hash(char *key, int key_len) {
47  unsigned int hash = 0;
48  int index;
49 
50  for (index = 0; index < key_len; index++) {
51  unsigned char c = key[index];
52  hash += c;
53  hash += (hash << 10);
54  hash ^= (hash >> 6);
55  }
56  hash += (hash << 3);
57  hash ^= (hash >> 11);
58  hash += (hash << 15);
59 
60  return hash;
61 }

63 void dictionary_put(t_dictionary *self, char *key, void *data) {
64  t_hash_element *existing_element = dictionary_get_element(self, key);
65 
66  if(existing_element != NULL) {
67  existing_element->data = data;
68  return;
69  }
70 
71  unsigned int key_hash = dictionary_hash(key, strlen(key));
72  int index = key_hash % self->table_max_size;
73  t_hash_element * new_element = dictionary_create_element(strdup(key), key_hash, data);
74 
75  t_hash_element *element = self->elements[index];
76 
77  if (element == NULL) {
78  self->elements[index] = new_element;
79  self->table_current_size++;
80 
81  if (self->table_current_size >= self->table_max_size) {
82  dictionary_resize(self, self->table_max_size * 2);
83  }
84 
85  } else {
86 
87  while (element->next != NULL) {
88  element = element->next;
89  }
90 
91  element->next = new_element;
92  }
93 
94  self->elements_amount++;
95 }

186 static void dictionary_resize(t_dictionary *self, int new_max_size) {
187  t_hash_element **new_table = calloc(new_max_size, sizeof(t_hash_element*));
188  t_hash_element **old_table = self->elements;
189 
190  self->table_current_size = 0;
191 
192  int table_index;
193  for (table_index = 0; table_index < self->table_max_size; table_index++) {
194  t_hash_element *old_element = old_table[table_index];
195  t_hash_element *next_element;
196 
197  while( old_element != NULL) {
198  // new position
199  int new_index = old_element->hashcode % new_max_size;
200  t_hash_element *new_element = new_table[new_index];
201 
202  if (new_element == NULL) {
203  new_table[new_index] = old_element;
204  self->table_current_size++;
205  } else {
206 
207  while (new_element->next != NULL) {
208  new_element = new_element->next;
209  }
210 
211  new_element->next = old_element;
212  }
213 
214  next_element = old_element->next;
215  old_element->next = NULL;
216  old_element = next_element;
217  }
218  }
219 
220  self->elements = new_table;
221  self->table_max_size = new_max_size;
222  free(old_table);
223 }

259 static t_hash_element *dictionary_get_element(t_dictionary *self, char *key) {
260  unsigned int key_hash = dictionary_hash(key, strlen(key));
261  int index = key_hash % self->table_max_size;
262 
263  t_hash_element *element = self->elements[index];
264 
265  if (element == NULL) {
266  return NULL;
267  }
268 
269  do {
270 
271  if (element->hashcode == key_hash) {
272  return element;
273  }
274 
275  } while ((element = element->next) != NULL);
276 
277  return NULL;
278 }

280 static void *dictionary_remove_element(t_dictionary *self, char *key) {
281  unsigned int key_hash = dictionary_hash(key, strlen(key));
282  int index = key_hash % self->table_max_size;
283 
284  t_hash_element *element = self->elements[index];
285 
286  if (element == NULL) {
287  return NULL;
288  }
289 
290  if (element->hashcode == key_hash) {
291  void *data = element->data;
292  self->elements[index] = element->next;
293  if (self->elements[index] == NULL) {
294  self->table_current_size--;
295  }
296  free(element->key);
297  free(element);
298  return data;
299  }
300 
301  while (element->next != NULL) {
302 
303  if (element->next->hashcode == key_hash) {
304  void *data = element->next->data;
305  t_hash_element *aux = element->next;
306  element->next = element->next->next;
307  free(aux->key);
308  free(aux);
309  return data;
310  }
311 
312  element = element->next;
313  }
314 
315  return NULL;
316 }
