26 typedef struct {
27  char *name;
28  unsigned char age;
29 } t_person;
30 
31 static t_person *persona_create(char *name, unsigned char age) {
32  t_person *new = malloc(sizeof(t_person));
33  new->name = strdup(name);
34  new->age = age;
35  return new;
36 }
37 
38 static void persona_destroy(t_person *self) {
39  free(self->name);
40  free(self);
41 }

43 static bool _ayudantes_menor(t_person *joven, t_person *menos_joven) {
44  return joven->age <= menos_joven->age;
45 }
46 
47 static bool _ayudantes_alfabetico(t_person *primero, t_person *segundo) {
48  return strcmp(primero->name, segundo->name) <= 0;
49 }
50 
51 static void* _ayudantes_minimo_edad(t_person* person1, t_person* person2) {
52  return person1->age <= person2->age ? person1 : person2;
53 }
54 
55 static void* _ayudantes_maximo_edad(t_person* person1, t_person* person2) {
56  return person1->age >= person2->age ? person1 : person2;
57 }
