303  va_list copy_arguments;

305  size_t buffer_size = vsnprintf(NULL, 0, format, copy_arguments) + 1;

316 char** _string_split(char* text, char* separator, bool(*is_last_token)(int)) {
317  char **substrings = string_array_new();
318  int index = 0;
319 
320  char *start = text;
321  char *end;
322 
323  while (_string_match(start, separator, &end) && !is_last_token(index)) {
324  _string_array_push(&substrings, string_substring_until(start, end - start), index++);
325  start = end + string_length(separator);
326  }
327 
328  _string_array_push(&substrings, string_duplicate(start), index);
329  return substrings;
330 }
