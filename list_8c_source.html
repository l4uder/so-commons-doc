261 void list_sort(t_list *self, bool (*comparator)(void *, void *)) {
262  if (list_size(self) < 2) {
263  return;
264  }
265 
266  t_list *left = list_take_and_remove(self, list_size(self) / 2);
267  list_sort(left, comparator);
268  t_list_iterator left_iterator = list_iterator_init(left);
269 
270  t_list *right = self;
271  list_sort(right, comparator);
272  t_list_iterator right_iterator = list_iterator_init(right);
273 
274  while (list_iterator_has_next(&left_iterator)) {
275  t_link_element *element = list_iterator_remove_element(&left_iterator);
276  list_iterator_add_element_sorted(&right_iterator, element, comparator);
277  list_iterator_advance(&right_iterator);
278  }
279 
280  list_destroy(left);
281 }
