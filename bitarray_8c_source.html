20 /* array index for character containing bit */
21 char _bit_in_char(int bit, bit_numbering_t mode){
22  switch(mode){
23  case LSB_FIRST:
24  return (0x80 >> (CHAR_BIT - 1 - (bit % CHAR_BIT)));
25  case MSB_FIRST:
26  return (0x80 >> (bit % CHAR_BIT));
27  default:
28  return _bit_in_char(bit, LSB_FIRST);
29  }
30 }
