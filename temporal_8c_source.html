26 char *temporal_get_string_time(const char* format) {
27  char* str_time = strdup(format);
28 
29  struct timespec log_timespec;
30  struct tm log_tm;
31  char* milisec;
32 
33  if (clock_gettime(CLOCK_REALTIME, &log_timespec) == -1) {
34  error_show("Error getting date!");
35  free(str_time);
36  return NULL;
37  }
38  milisec = string_from_format("%03ld", log_timespec.tv_nsec / 1000000);
39 
40  for (char* ms = strstr(str_time, "%MS"); ms != NULL; ms = strstr(ms + 3, "%MS")) {
41  memcpy(ms, milisec, 3);
42  }
43 
44  localtime_r(&log_timespec.tv_sec, &log_tm);
45  strftime(str_time, strlen(format) + 1, str_time, &log_tm);
46 
47  free(milisec);
48 
49  return str_time;
50 }

106 static int64_t calculate_delta_ms(t_temporal* temporal) {
107  struct timespec now;
108  if (clock_gettime(CLOCK_MONOTONIC_RAW, &now) == -1) {
109  error_show("Error getting time!");
110  return 0;
111  }
112 
113  return (now.tv_sec - temporal->current.tv_sec) * 1000
114  + (now.tv_nsec - temporal->current.tv_nsec) / 1000000;
115 }
